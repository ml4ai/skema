38class CastToAnnotatedCastVisitor: 

39 """ 

40 class CastToAnnotatedCastVisitor - A visitor that traverses CAST nodes 

41 and generates an annotated cast version of the CAST. 

42 

43 The AnnCastNodes have additional attributes (fields) that are used 

44 in a later pass to maintain scoping information for GrFN containers. 

45 """

53 def generate_annotated_cast(self, grfn_2_2: bool = False): 

54 nodes = self.cast.nodes 

55 

56 annotated_cast = [] 

57 for node in nodes: 

58 annotated_cast.append(self.visit(node)) 

59 

60 return PipelineState(annotated_cast, grfn_2_2)
