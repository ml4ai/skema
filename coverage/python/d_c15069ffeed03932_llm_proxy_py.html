40async def get_lines_of_model(zip_file: UploadFile = File()) -> Dynamics: 

41 """ 

42 Endpoint for generating a line span containing the dynamics from a zip archive. Currently 

43 it only expects there to be one python file in the zip. There can be other files, such as a 

44 README.md, but only one .py. Future versions will generalize support to arbritary zip contents.  

45 

46 ### Python example 

47 ``` 

48 import requests 

49 

50 files = { 

51 "zip_file": open(zip_path, "rb"), 

52 } 

53 

54 response = requests.post(f"{ENDPOINT}/morae/linespan-given-filepaths-zip", files=files) 

55 gromet_json = response.json() 

56 """ 

57 files=[] 

58 blobs=[] 

59 block=[] 

60 with ZipFile(BytesIO(zip_file.file.read()), "r") as zip: 

61 for file in zip.namelist(): 

62 file_obj = Path(file) 

63 if file_obj.suffix in [".py"]: 

64 files.append(file) 

65 blobs.append(zip.open(file).read()) 

66 

67 # read in the code, for the prompt 

68 code = blobs[0].decode("utf-8") # needs to be regular string, not byte string 

69 file = files[0] 

70 # json for the fn construction 

71 single_snippet_payload = { 

72 "files": [file], 

73 "blobs": [code], 

74 } 

75 

76 # this is the formatting instructions 

77 response_schemas = [ 

78 ResponseSchema(name="model_function", description="The name of the function that contains the model dynamics") 

79 ] 

80 

81 # for structured output parsing, converts schema to langhchain object 

82 output_parser = StructuredOutputParser.from_response_schemas(response_schemas) 

83 

84 # for structured output parsing, makes the instructions to be passed as a variable to prompt template 

85 format_instructions = output_parser.get_format_instructions() 

86 

87 # low temp as is not generative 

88 temperature = 0.1 

89 

90 # initialize the models 

91 openai = ChatOpenAI( 

92 temperature=temperature, 

93 model_name='gpt-3.5-turbo', 

94 openai_api_key=SKEMA_OPENAI_KEY 

95 ) 

96 

97 # construct the prompts 

98 template="You are a assistant that answers questions about code." 

99 system_message_prompt = SystemMessagePromptTemplate.from_template(template) 

100 human_template="Find the function that contains the model dynamics in {code} \n{format_instructions}" 

101 human_message_prompt = HumanMessagePromptTemplate.from_template(human_template) 

102 

103 # combining the templates for a chat template 

104 chat_prompt = ChatPromptTemplate.from_messages([system_message_prompt, human_message_prompt]) 

105 

106 # formatting the prompt with input variables 

107 formatted_prompt = chat_prompt.format_prompt(code=code, format_instructions = format_instructions).to_messages() 

108 

109 # running the model 

110 output = openai(formatted_prompt) 

111 

112 # parsing the output 

113 try: 

114 parsed_output = output_parser.parse(output.content) 

115 

116 function_name = parsed_output['model_function'] 

117 

118 # Get the FN from it 

119 url = "https://api.askem.lum.ai/code2fn/fn-given-filepaths" 

120 response_zip = requests.post(url, json=single_snippet_payload) 

121 

122 # get metadata entry for function 

123 for entry in response_zip.json()['modules'][0]['fn_array']: 

124 try: 

125 if entry['b'][0]['name'][0:len(function_name)] == function_name: 

126 metadata_idx = entry['b'][0]['metadata'] 

127 except: 

128 None 

129 

130 # get line span using metadata 

131 for (i,metadata) in enumerate(response_zip.json()['modules'][0]['metadata_collection']): 

132 if i == (metadata_idx - 1): 

133 line_begin = metadata[0]['line_begin'] 

134 line_end = metadata[0]['line_end'] 

135 except: 

136 print("Failed to parse dynamics") 

137 line_begin = 0 

138 line_end = 0 

139 

140 block.append(f"L{line_begin}-L{line_end}") 

141 

142 output = Dynamics(name=None, description=None, block=block) 

143 return output
