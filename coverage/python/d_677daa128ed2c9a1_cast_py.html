71def compare_name_nodes(name1: Name, name2: Name) -> bool: 

72 """ 

73 Checks if two Name nodes are equal, by only looking at 

74 their `name` fields. The `__eq__` method on `Name` nodes generated 

75 by Swagger also checks the `id` attribute, which we do not expect 

76 to be consistent across CAST generations, since it is a UUID 

77 """ 

78 if not (isinstance(name1, Name) and isinstance(name2, Name)): 

79 return False 

80 return name1.name == name2.name

104 def __eq__(self, other): 

105 """ 

106 For equality, the two CAST objects must have the same node data. 

107 When checking each node, we allow a custom node comparison function. 

108 Currently, the only case where we do a custom comparison is for Name nodes. 

109 For all other nodes we use their Swagger generated `__eq__` method 

110 """ 

111 if len(self.nodes) != len(other.nodes): 

112 return False 

113 

114 for i, node in enumerate(self.nodes): 

115 other_node = other.nodes[i] 

116 if isinstance(node, Name): 

117 comparator = compare_name_nodes 

118 else: 

119 comparator = lambda n1, n2: n1 == n2 

120 

121 if not comparator(node, other_node): 

122 # printing diff to help locating difference 

123 print(f"CAST __eq__ failed:") 

124 self_lines = str(node).splitlines() 

125 other_lines = str(other_node).splitlines() 

126 for i, diff in enumerate( 

127 difflib.ndiff(self_lines, other_lines) 

128 ): 

129 if diff[0] == " ": 

130 continue 

131 print(f"Line {i}: {diff}") 

132 return False 

133 

134 return True

195 def to_AIR(self): 

196 air = self.to_air_dict() 

197 

198 C, V, T, D = dict(), dict(), dict(), dict() 

199 

200 # Create variable definitions 

201 for var_data in air["variables"]: 

202 new_var = VariableDefinition.from_data(var_data) 

203 V[new_var.identifier] = new_var 

204 

205 # Create type definitions 

206 for type_data in air["types"]: 

207 new_type = TypeDefinition.from_dict(type_data) 

208 T[new_type.identifier] = new_type 

209 

210 # Create container definitions 

211 for con_data in air["containers"]: 

212 new_container = GenericContainer.from_dict(con_data) 

213 for in_var in new_container.arguments: 

214 if in_var not in V: 

215 V[in_var] = VariableDefinition.from_identifier(in_var) 

216 C[new_container.identifier] = new_container 

217 

218 return AutoMATES_IR( 

219 GenericIdentifier.from_str(air["entrypoint"]), C, V, T, [], [], []

259 @classmethod 

260 def parse_cast_json(cls, data): 

261 if isinstance(data, list): 

262 # If we see a list parse each one of its elements 

263 return [cls.parse_cast_json(item) for item in data] 

264 elif data is None: 

265 return None 

266 elif isinstance(data, (float, int, str, bool)): 

267 # If we see a primitave type, simply return its value 

268 return data 

269 elif all( 

270 k in data for k in ("row_start", "row_end", "col_start", "col_end") 

271 ): 

272 return SourceRef( 

273 row_start=data["row_start"], 

274 row_end=data["row_end"], 

275 col_start=data["col_start"], 

276 col_end=data["col_end"], 

277 source_file_name=data["source_file_name"], 

278 ) 

279 

280 if "node_type" in data: 

281 # Create the object specified by "node_type" object with the values 

282 # from its children nodes 

283 for node_type in CAST_NODES_TYPES_LIST: 

284 

285 if node_type.__name__ == data["node_type"]: 

286 node_results = { 

287 k: cls.parse_cast_json(v) 

288 for k, v in data.items() 

289 if k != "node_type" 

290 } 

291 return node_type(**node_results) 

292 

293 raise CASTJsonException( 

294 f"Unable to decode json CAST field with field names: {set(data.keys())}" 

295 )
