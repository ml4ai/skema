25async def equations_to_amr(data: schema.EquationImagesToAMR): 

26 """ 

27 Converts images of equations to AMR. 

28 

29 ### Python example 

30 ``` 

31 from pathlib import Path 

32 import base64 

33 import requests 

34 

35 

36 with Path("bayes-rule-white-bg.png").open("rb") as infile: 

37 img_bytes = infile.read() 

38 img_b64 = base64.b64encode(img_bytes).decode("utf-8") 

39 r = requests.post(url, data=img_b64) 

40 print(r.text) 

41 

42 images_bytes = [open("eq1.png", "rb").read(), open("eq2.png", "rb").read()] 

43 

44 images_b64 = [base64.b64encode(img_bytes).decode("utf-8") for img_bytes in images_bytes] 

45 

46 url = "0.0.0.0" 

47 r = requests.post(f"{url}/workflows/images/base64/equations-to-amr", json={"images": images_b64, "model": "regnet"}) 

48 r.json() 

49 """ 

50 mml: List[str] = [utils.clean_mml(eqn2mml.b64_image_to_mml(img)) for img in data.images] 

51 payload = {"mathml": mml, "model": data.model} 

52 # FIXME: why is this a PUT?

54 if res.status_code != 200: 

55 return JSONResponse( 

56 status_code=400, 

57 content={ 

58 "error": f"MORAE PUT /mathml/amr failed to process payload", 

59 "payload": payload, 

60 },

67async def equations_to_amr(data: schema.EquationLatexToAMR): 

68 """ 

69 Converts equations (in LaTeX) to AMR. 

70 

71 ### Python example 

72 ``` 

73 import requests 

74 

75 equations = [ 

76 "\\frac{\\delta x}{\\delta t} = {\\alpha x} - {\\beta x y}", 

77 "\\frac{\\delta y}{\\delta t} = {\\alpha x y} - {\\gamma y}" 

78 ] 

79 url = "0.0.0.0" 

80 r = requests.post(f"{url}/workflows/latex/equations-to-amr", json={"equations": equations, "model": "regnet"}) 

81 r.json() 

82 """ 

83 mml: List[str] = [ 

84 utils.clean_mml(eqn2mml.get_mathml_from_latex(tex)) for tex in data.equations 

85 ] 

86 payload = {"mathml": mml, "model": data.model} 

87 res = requests.put(f"{SKEMA_RS_ADDESS}/mathml/amr", json=payload) 

88 if res.status_code != 200: 

89 return JSONResponse( 

90 status_code=400, 

91 content={ 

92 "error": f"MORAE PUT /mathml/amr failed to process payload with error {res.text}", 

93 "payload": payload,
